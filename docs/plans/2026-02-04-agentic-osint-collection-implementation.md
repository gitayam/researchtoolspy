```sql
CREATE TABLE IF NOT EXISTS collection_jobs (
  id TEXT PRIMARY KEY,
  workspace_id TEXT NOT NULL,
  query TEXT NOT NULL,
  categories TEXT DEFAULT '["news","academic","government","social","technical"]', -- JSON array
  time_range TEXT DEFAULT 'year', -- day, week, month, year, all

  status TEXT DEFAULT 'pending', -- pending, running, complete, error
  results_count INTEGER DEFAULT 0,
  batch_job_id TEXT, -- Links to analysis batch job after approval

  created_at TEXT NOT NULL DEFAULT (datetime('now')),
  completed_at TEXT,
  FOREIGN KEY (workspace_id) REFERENCES workspaces(id)
);

-- Individual search results for triage
CREATE TABLE IF NOT EXISTS collection_results (
  id TEXT PRIMARY KEY,
  job_id TEXT NOT NULL,
  url TEXT NOT NULL,
  title TEXT,
  snippet TEXT,
  category TEXT, -- news, academic, government, social, technical
  source_domain TEXT,
  relevance_score INTEGER DEFAULT 50, -- 0-100

  engine TEXT, -- Which SearXNG engine found it
```

```typescript
import type { PagesFunction } from '@cloudflare/workers-types';

interface Env {
  DB: D1Database;
  CACHE: KVNamespace;
  OSINT_AGENT_URL?: string;
}

interface CollectionRequest {
  query: string;
  categories?: string[];
  timeRange?: string;
  maxResults?: number;
  useLocalLLM?: boolean;
  workspaceId?: string;
}

export const onRequestPost: PagesFunction<Env> = async (context) => {
  try {
    const body = await context.request.json() as CollectionRequest;
    const { query, categories, timeRange, maxResults, useLocalLLM, workspaceId } = body;

    if (!query || query.trim().length < 3) {
      return Response.json({ error: 'Query must be at least 3 characters' }, { status: 400 });
    }

    // Generate job ID
    const jobId = crypto.randomUUID();
    const defaultCategories = ['news', 'academic', 'government', 'social', 'technical'];
    const finalCategories = categories && categories.length > 0 ? categories : defaultCategories;

    // Insert job record
    await context.env.DB.prepare(`
      INSERT INTO collection_jobs (id, workspace_id, query, categories, time_range, max_results, status, created_at)
      VALUES (?, ?, ?, ?, ?, ?, 'pending', datetime('now'))
    `).bind(
      jobId,
      workspaceId || 'default',
      query.trim(),
      JSON.stringify(finalCategories),
      timeRange || 'year',
      maxResults || 100
    ).run();

    // Get agent URL from env or use default
    const agentUrl = context.env.OSINT_AGENT_URL || 'http://osint-agent:8000';
    const callbackUrl = new URL('/api/collection/callback', context.request.url).toString();

    // Trigger agent collection (fire and forget)
    const agentRequest = {
      jobId,
      query: query.trim(),
      categories: finalCategories,
      maxResults: maxResults || 100,
      timeRange: timeRange || 'year',
      searxngEndpoint: 'http://searxng:8080',
      callbackUrl,
      useLocalLLM: useLocalLLM || false
    };

    // Update status to running
    await context.env.DB.prepare(`
      UPDATE collection_jobs SET status = 'running' WHERE id = ?
    `).bind(jobId).run();

    // Fire async request to agent (don't await)
    context.waitUntil(
      fetch(`${agentUrl}/collect`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(agentRequest)
      }).catch(async (error) => {
        // Update job status on connection failure
        await context.env.DB.prepare(`
          UPDATE collection_jobs SET status = 'error', error_message = ? WHERE id = ?
        `).bind(`Agent connection failed: ${error.message}`, jobId).run();
      })
    );

    return Response.json({
      jobId,
      status: 'started',
      message: 'Collection job initiated'
    });

  } catch (error: any) {
    console.error('Collection start error:', error);
    return Response.json({ error: error.message || 'Failed to start collection' }, { status: 500 });
  }
};
```

```typescript
export const onRequestGet: PagesFunction<Env> = async (context) => {
  try {
    const jobId = context.params.jobId as string;

    if (!jobId) {
      return Response.json({ error: 'Job ID required' }, { status: 400 });
    }

    // Get job details
    const job = await context.env.DB.prepare(`
      SELECT id, workspace_id, query, categories, time_range, max_results,
             status, results_count, batch_job_id, error_message, llm_used,
             created_at, completed_at
      FROM collection_jobs WHERE id = ?
    `).bind(jobId).first();

    if (!job) {
      return Response.json({ error: 'Job not found' }, { status: 404 });
    }

    // Parse categories JSON
    const parsedJob = {
      ...job,
      categories: JSON.parse(job.categories as string || '[]')
    };

    // If complete, include results summary
    if (job.status === 'complete') {
      const summary = await context.env.DB.prepare(`
        SELECT category, COUNT(*) as count, AVG(relevance_score) as avg_relevance
        FROM collection_results WHERE job_id = ?
        GROUP BY category
        ORDER BY count DESC
      `).bind(jobId).all();

      return Response.json({
        ...parsedJob,
        resultsSummary: summary.results
      });
    }

    return Response.json(parsedJob);

  } catch (error: any) {
    console.error('Collection status error:', error);
    return Response.json({ error: error.message || 'Failed to get status' }, { status: 500 });
  }
};
```

```typescript
import type { PagesFunction } from '@cloudflare/workers-types';

interface Env {
  DB: D1Database;
}

export const onRequestGet: PagesFunction<Env> = async (context) => {
  try {
    const jobId = context.params.jobId as string;
    const url = new URL(context.request.url);

    const category = url.searchParams.get('category');
    const minRelevance = parseInt(url.searchParams.get('minRelevance') || '0');
    const approved = url.searchParams.get('approved'); // 'pending', 'approved', 'rejected', 'all'
    const limit = Math.min(parseInt(url.searchParams.get('limit') || '50'), 200);
    const offset = parseInt(url.searchParams.get('offset') || '0');

    if (!jobId) {
      return Response.json({ error: 'Job ID required' }, { status: 400 });
    }

    // Build query dynamically
    let query = `
      SELECT id, job_id, url, title, snippet, category, source_domain,
             relevance_score, published_date, engine, approved, approved_at,
             analysis_id, created_at
      FROM collection_results
      WHERE job_id = ? AND relevance_score >= ?
    `;
    const params: any[] = [jobId, minRelevance];

    if (category) {
      query += ' AND category = ?';
      params.push(category);
    }

    if (approved && approved !== 'all') {
      const approvalValue = approved === 'approved' ? 1 : approved === 'rejected' ? -1 : 0;
      query += ' AND approved = ?';
      params.push(approvalValue);
    }

    query += ' ORDER BY relevance_score DESC LIMIT ? OFFSET ?';
    params.push(limit, offset);

    const results = await context.env.DB.prepare(query).bind(...params).all();

    // Get total count for pagination
    let countQuery = `
      SELECT COUNT(*) as total FROM collection_results
      WHERE job_id = ? AND relevance_score >= ?
    `;
    const countParams: any[] = [jobId, minRelevance];

    if (category) {
      countQuery += ' AND category = ?';
      countParams.push(category);
    }
    if (approved && approved !== 'all') {
      const approvalValue = approved === 'approved' ? 1 : approved === 'rejected' ? -1 : 0;
      countQuery += ' AND approved = ?';
      countParams.push(approvalValue);
    }

    const countResult = await context.env.DB.prepare(countQuery).bind(...countParams).first();
    const total = (countResult?.total as number) || 0;

    return Response.json({
      results: results.results,
      pagination: {
        limit,
        offset,
        total,
        hasMore: offset + results.results.length < total
      }
    });

  } catch (error: any) {
    console.error('Collection results error:', error);
    return Response.json({ error: error.message || 'Failed to get results' }, { status: 500 });
  }
};
```

```typescript
import type { PagesFunction } from '@cloudflare/workers-types';

interface Env {
  DB: D1Database;
}

interface ApproveRequest {
  selectedIds: string[];
  analyzeNow?: boolean;
}

export const onRequestPost: PagesFunction<Env> = async (context) => {
  try {
    const jobId = context.params.jobId as string;
    const body = await context.request.json() as ApproveRequest;
    const { selectedIds, analyzeNow = true } = body;

    if (!jobId) {
      return Response.json({ error: 'Job ID required' }, { status: 400 });
    }

    if (!selectedIds || selectedIds.length === 0) {
      return Response.json({ error: 'No results selected' }, { status: 400 });
    }

    // Mark selected results as approved
    const placeholders = selectedIds.map(() => '?').join(',');
    await context.env.DB.prepare(`
      UPDATE collection_results
      SET approved = 1, approved_at = datetime('now')
      WHERE job_id = ? AND id IN (${placeholders})
    `).bind(jobId, ...selectedIds).run();

    if (analyzeNow) {
      // Get URLs for approved results
      const approved = await context.env.DB.prepare(`
        SELECT id, url FROM collection_results
        WHERE job_id = ? AND id IN (${placeholders})
      `).bind(jobId, ...selectedIds).all();

      const urls = approved.results.map((r: any) => r.url);

      if (urls.length === 0) {
        return Response.json({ error: 'No valid URLs found' }, { status: 400 });
      }

      // Create batch items for existing batch processor
      const batchItems = urls.map((url: string, index: number) => ({
        id: `item-${index}`,
        type: 'url',
        source: url
      }));

      // Trigger batch analysis using existing endpoint
      const origin = new URL(context.request.url).origin;
      const batchResponse = await fetch(`${origin}/api/tools/batch-process`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          operation: 'analyze-url',
          items: batchItems,
          options: { maxWorkers: 3, retryFailed: true }
        })
      });

      if (!batchResponse.ok) {
        const errorData = await batchResponse.json();
        return Response.json({
          error: 'Batch analysis failed to start',
          details: errorData
        }, { status: 500 });
      }

      const batchJob = await batchResponse.json();

      // Link batch job to collection job
      await context.env.DB.prepare(`
        UPDATE collection_jobs SET batch_job_id = ? WHERE id = ?
      `).bind(batchJob.jobId || 'batch-' + Date.now(), jobId).run();

      return Response.json({
        approved: selectedIds.length,
        batchJobId: batchJob.jobId,
        status: 'analysis_started',
        urls: urls.length
      });
    }

    return Response.json({
      approved: selectedIds.length,
      status: 'saved'
    });

  } catch (error: any) {
    console.error('Collection approve error:', error);
    return Response.json({ error: error.message || 'Failed to approve results' }, { status: 500 });
  }
};

// Also handle rejections
export const onRequestDelete: PagesFunction<Env> = async (context) => {
  try {
    const jobId = context.params.jobId as string;
    const body = await context.request.json() as { selectedIds: string[] };
    const { selectedIds } = body;

    if (!jobId || !selectedIds || selectedIds.length === 0) {
      return Response.json({ error: 'Job ID and selected IDs required' }, { status: 400 });
    }

    const placeholders = selectedIds.map(() => '?').join(',');
    await context.env.DB.prepare(`
      UPDATE collection_results
      SET approved = -1, approved_at = datetime('now')
      WHERE job_id = ? AND id IN (${placeholders})
    `).bind(jobId, ...selectedIds).run();

    return Response.json({
      rejected: selectedIds.length,
      status: 'rejected'
    });

  } catch (error: any) {
    console.error('Collection reject error:', error);
    return Response.json({ error: error.message || 'Failed to reject results' }, { status: 500 });
  }
};
```

```typescript
import type { PagesFunction } from '@cloudflare/workers-types';

interface Env {
  DB: D1Database;
}

interface AgentCallback {
  jobId: string;
  status: 'complete' | 'error';
  results?: Array<{
    url: string;
    title: string;
    snippet: string;
    category: string;
    source_domain: string;
    relevance_score: number;
    published_date?: string;
    engine: string;
  }>;
  queries?: Array<{
    category: string;
    query: string;
    rationale: string;
    results_count: number;
  }>;
  error?: string;
  llm_used?: string;
}

export const onRequestPost: PagesFunction<Env> = async (context) => {
  try {
    const body = await context.request.json() as AgentCallback;
    const { jobId, status, results, queries, error, llm_used } = body;

    if (!jobId) {
      return Response.json({ error: 'Job ID required' }, { status: 400 });
    }

    // Verify job exists
    const job = await context.env.DB.prepare(`
      SELECT id, status FROM collection_jobs WHERE id = ?
    `).bind(jobId).first();

    if (!job) {
      return Response.json({ error: 'Job not found' }, { status: 404 });
    }

    if (status === 'error') {
      // Update job as failed
      await context.env.DB.prepare(`
        UPDATE collection_jobs
        SET status = 'error', error_message = ?, completed_at = datetime('now')
        WHERE id = ?
      `).bind(error || 'Unknown error', jobId).run();

      return Response.json({ received: true, status: 'error_recorded' });
    }

    // Insert results
    if (results && results.length > 0) {
      const insertStmt = context.env.DB.prepare(`
        INSERT INTO collection_results (id, job_id, url, title, snippet, category, source_domain, relevance_score, published_date, engine, created_at)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, datetime('now'))
      `);

      // Batch insert results
      const batch = results.map(r =>
        insertStmt.bind(
          crypto.randomUUID(),
          jobId,
          r.url,
          r.title || '',
          r.snippet || '',
          r.category || 'general',
          r.source_domain || '',
          r.relevance_score || 50,
          r.published_date || null,
          r.engine || 'unknown'
        )
      );

      await context.env.DB.batch(batch);
    }

    // Insert queries
    if (queries && queries.length > 0) {
      const queryStmt = context.env.DB.prepare(`
        INSERT INTO collection_queries (id, job_id, category, query, rationale, results_count, created_at)
        VALUES (?, ?, ?, ?, ?, ?, datetime('now'))
      `);

      const queryBatch = queries.map(q =>
        queryStmt.bind(
          crypto.randomUUID(),
          jobId,
          q.category,
          q.query,
          q.rationale || '',
          q.results_count || 0
        )
      );

      await context.env.DB.batch(queryBatch);
    }

    // Update job as complete
    await context.env.DB.prepare(`
      UPDATE collection_jobs
      SET status = 'complete', results_count = ?, llm_used = ?, completed_at = datetime('now')
      WHERE id = ?
    `).bind(results?.length || 0, llm_used || 'unknown', jobId).run();

    return Response.json({
      received: true,
      status: 'complete',
      resultsStored: results?.length || 0,
      queriesStored: queries?.length || 0
    });

  } catch (error: any) {
    console.error('Collection callback error:', error);
    return Response.json({ error: error.message || 'Callback failed' }, { status: 500 });
  }
};
```

# containers/osint-agent/main.py
from fastapi import FastAPI, BackgroundTasks
from pydantic import BaseModel
from typing import Optional
import httpx
import asyncio
import os
import json

app = FastAPI(title="OSINT Collection Agent")

# ─────────────────────────────────────────────────────────────────────────
# LLM CONFIGURATION
# ─────────────────────────────────────────────────────────────────────────

def get_openai_client():
    """Get OpenAI client for query expansion and scoring"""
    from openai import OpenAI
    return OpenAI(api_key=os.getenv("OPENAI_API_KEY"))

def get_local_client():
    """Get local LLM client (Ollama-compatible)"""
    from openai import OpenAI
    return OpenAI(
        base_url=os.getenv("LOCAL_AI_URL", "http://localhost:11434/v1"),
        api_key=os.getenv("LOCAL_AI_API_KEY", "ollama")
    )

# ─────────────────────────────────────────────────────────────────────────
# REQUEST MODELS
# ─────────────────────────────────────────────────────────────────────────

class CollectionRequest(BaseModel):
    jobId: str
    query: str
    categories: list[str]
    maxResults: int = 100
    timeRange: str = "year"
    searxngEndpoint: str = "http://searxng:8080"
    callbackUrl: str
    useLocalLLM: bool = False

# ─────────────────────────────────────────────────────────────────────────
# QUERY EXPANSION
# ─────────────────────────────────────────────────────────────────────────

async def expand_queries(query: str, categories: list[str], use_local: bool = False) -> list[dict]:
    """Use LLM to expand research question into targeted search queries"""

    client = get_local_client() if use_local else get_openai_client()
    model = os.getenv("LOCAL_AI_MODEL", "irregularbot:latest") if use_local else "gpt-5"

    prompt = f"""You are an OSINT research specialist. Given a research question, generate targeted search queries for each category.

Research Question: {query}
Categories: {', '.join(categories)}

For each category, generate 2-3 specific search queries that would find relevant sources. Use:
- Specific names, terms, and keywords
- Date ranges where relevant (e.g., "2023..2024")
- Search operators (quotes for exact phrases)

Output as JSON array:
[{{"category": "news", "query": "specific search query", "rationale": "why this query"}}]

Only output valid JSON, no other text."""

    try:
        response = client.chat.completions.create(
            model=model,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.7,
            max_tokens=2000
        )

        content = response.choices[0].message.content.strip()
        # Try to parse JSON from response
        if content.startswith('['):
            return json.loads(content)
        # Try to extract JSON from markdown code block
        if '```json' in content:
            content = content.split('```json')[1].split('```')[0].strip()
            return json.loads(content)
        if '```' in content:
            content = content.split('```')[1].split('```')[0].strip()
            return json.loads(content)
        return json.loads(content)
    except Exception as e:
        print(f"Query expansion error: {e}")
        # Fallback: simple queries per category
        return [{"category": cat, "query": query, "rationale": "fallback"} for cat in categories]

# ─────────────────────────────────────────────────────────────────────────
# SEARCH EXECUTION
# ─────────────────────────────────────────────────────────────────────────

async def execute_searches(queries: list[dict], searxng_endpoint: str, max_results: int) -> list[dict]:
    """Execute searches against SearXNG"""

    results = []
    seen_urls = set()

    async with httpx.AsyncClient(timeout=30.0) as client:
        for q in queries:
            try:
                response = await client.get(
                    f"{searxng_endpoint}/search",
                    params={
                        "q": q["query"],
                        "categories": q["category"],
                        "format": "json"
                    }
                )

                if response.status_code == 200:
                    data = response.json()
                    for r in data.get("results", []):
                        url = r.get("url", "")
                        if url and url not in seen_urls:
                            seen_urls.add(url)
                            results.append({
                                "url": url,
                                "title": r.get("title", ""),
                                "snippet": r.get("content", "")[:500],
                                "category": q["category"],
                                "source_domain": extract_domain(url),
                                "relevance_score": 50,  # Will be updated by scoring
                                "published_date": r.get("publishedDate"),
                                "engine": r.get("engine", "unknown")
                            })

                # Rate limiting
                await asyncio.sleep(0.5)

            except Exception as e:
                print(f"Search error for {q['query']}: {e}")
                continue

    return results[:max_results]

def extract_domain(url: str) -> str:
    """Extract domain from URL"""
    from urllib.parse import urlparse
    try:
        return urlparse(url).netloc
    except:
        return ""

# ─────────────────────────────────────────────────────────────────────────
# RELEVANCE SCORING
# ─────────────────────────────────────────────────────────────────────────

async def score_results(results: list[dict], query: str, use_local: bool = False) -> list[dict]:
    """Score results for relevance using LLM"""

    if not results:
        return results

    client = get_local_client() if use_local else get_openai_client()
    model = os.getenv("LOCAL_AI_MODEL", "irregularbot:latest") if use_local else "gpt-5-mini"

    # Score in batches of 30
    batch_size = 30
    scored = []

    for i in range(0, len(results), batch_size):
        batch = results[i:i+batch_size]

        # Format batch for scoring
        batch_text = "\n".join([
            f"{j+1}. [{r['category']}] {r['title']}\n   URL: {r['url']}\n   {r['snippet'][:200]}"
            for j, r in enumerate(batch)
        ])

        prompt = f"""Score each result's relevance (0-100) to the research question.

Research Question: {query}

Results to score:
{batch_text}

Consider:
- Title/snippet match to question (40%)
- Source authority/domain reputation (25%)
- Recency (20%)
- Content type appropriateness (15%)

Output as JSON array: [{{"index": 1, "score": 85}}, ...]
Only output valid JSON."""

        try:
            response = client.chat.completions.create(
                model=model,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3,
                max_tokens=1000
            )

            content = response.choices[0].message.content.strip()
            if '```' in content:
                content = content.split('```')[1].split('```')[0].strip()
                if content.startswith('json'):
                    content = content[4:].strip()

            scores = json.loads(content)
            score_map = {s["index"]: s["score"] for s in scores}

            for j, r in enumerate(batch):
                r["relevance_score"] = score_map.get(j+1, 50)
                scored.append(r)

        except Exception as e:
            print(f"Scoring error: {e}")
            scored.extend(batch)  # Keep default scores

    return scored

# ─────────────────────────────────────────────────────────────────────────
# MAIN ENDPOINT
# ─────────────────────────────────────────────────────────────────────────

@app.post("/collect")
async def collect(request: CollectionRequest, background_tasks: BackgroundTasks):
    """Start collection process"""
    background_tasks.add_task(run_collection, request)
    return {"status": "started", "jobId": request.jobId}

async def run_collection(request: CollectionRequest):
    """Full collection pipeline"""
    try:
        # Step 1: Expand queries
        queries = await expand_queries(
            request.query,
            request.categories,
            request.useLocalLLM
        )

        # Step 2: Execute searches
        results = await execute_searches(
            queries,
            request.searxngEndpoint,
            request.maxResults
        )

        # Step 3: Score results
        scored_results = await score_results(
            results,
            request.query,
            request.useLocalLLM
        )

        # Sort by relevance
        scored_results.sort(key=lambda x: x["relevance_score"], reverse=True)

        # Step 4: Callback with results
        async with httpx.AsyncClient(timeout=60.0) as client:
            await client.post(
                request.callbackUrl,
                json={
                    "jobId": request.jobId,
                    "status": "complete",
                    "results": scored_results,
                    "queries": [{"category": q["category"], "query": q["query"],
                                "rationale": q.get("rationale", ""),
                                "results_count": len([r for r in scored_results if r["category"] == q["category"]])}
                               for q in queries],
                    "llm_used": "local" if request.useLocalLLM else "openai"
                }
            )

    except Exception as e:
        print(f"Collection error: {e}")
        # Send error callback
        try:
            async with httpx.AsyncClient(timeout=30.0) as client:
                await client.post(
                    request.callbackUrl,
                    json={
                        "jobId": request.jobId,
                        "status": "error",
                        "error": str(e)
                    }
                )
        except:
            pass

# ─────────────────────────────────────────────────────────────────────────
# HEALTH & INFO ENDPOINTS
# ─────────────────────────────────────────────────────────────────────────

    return {

    }

    """Root endpoint"""
    return {"service": "OSINT Collection Agent", "version": "1.0.0"}

```tsx
            )}

            {jobStatus.error_message && (
              <p className="mt-4 text-sm text-red-500">{jobStatus.error_message}</p>
            )}
          </CardContent>
        </Card>
      )}

      {/* Triage Results */}
      {results?.results && results.results.length > 0 && (
        <Card>
          <CardHeader>
            <div className="flex items-center justify-between">
              <CardTitle className="flex items-center gap-2">
                <Database className="h-5 w-5" />
                Triage Results ({results.results.length})
              </CardTitle>
              <div className="flex items-center gap-2">
                <Button variant="outline" size="sm" onClick={() => selectAboveThreshold(70)}>
                  Select &gt;70%
                </Button>
                <Button variant="outline" size="sm" onClick={() => setSelectedResults(new Set())}>
                  Clear
                </Button>
              </div>
            </div>
            <CardDescription>
              Selected: {selectedResults.size} | Est. cost: ~${(selectedResults.size * 0.005).toFixed(2)}
            </CardDescription>
          </CardHeader>
          <CardContent>
            <div className="space-y-2 max-h-[500px] overflow-y-auto">
              {results.results.map((result: CollectionResult) => (
                <div
                  key={result.id}
                  className={`p-3 border rounded-lg flex items-start gap-3 hover:bg-muted/50 ${
                    selectedResults.has(result.id) ? 'border-primary bg-primary/5' : ''
                  }`}
                >
                  <Checkbox
                    checked={selectedResults.has(result.id)}
                    onCheckedChange={() => toggleResult(result.id)}
                  />
                  <div className="flex-1 min-w-0">
                    <div className="flex items-center gap-2 mb-1">
                      <Badge variant="outline">{result.relevance_score}%</Badge>
                      <Badge variant="secondary">{result.category}</Badge>
                      <span className="text-xs text-muted-foreground">{result.source_domain}</span>
                    </div>
                    <h4 className="font-medium truncate">{result.title || result.url}</h4>
                    <p className="text-sm text-muted-foreground line-clamp-2">{result.snippet}</p>
                  </div>
                  <a href={result.url} target="_blank" rel="noopener noreferrer">
                    <ExternalLink className="h-4 w-4 text-muted-foreground hover:text-primary" />
                  </a>
                </div>
              ))}
            </div>

            {selectedResults.size > 0 && (
              <div className="flex gap-2 mt-4 pt-4 border-t">
                <Button
                  onClick={() => approveResults.mutate(true)}
                  disabled={approveResults.isPending}
                >
                  {approveResults.isPending ? 'Processing...' : `Analyze ${selectedResults.size} Sources`}
                </Button>
                <Button
                  variant="outline"
                  onClick={() => approveResults.mutate(false)}
                  disabled={approveResults.isPending}
                >
                  Save to Library
                </Button>
              </div>
            )}
          </CardContent>
        </Card>
      )}
    </div>
  )
}
```

## Troubleshooting

**Container connection failures:**
- Verify containers are running: Check Cloudflare dashboard
- Check container logs for errors
- Verify environment variables are set

**Search returning no results:**
- SearXNG may be rate-limited by engines
- Try different search queries
- Check SearXNG logs for engine errors

**LLM errors:**
- Verify API keys are correct
- Check token limits aren't exceeded
- Try switching between OpenAI and local LLM
